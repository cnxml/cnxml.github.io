github多账号设置

1.touch ~/.ssh/config 2.vim 
    # default                                                                       
    Host github.com
    HostName github.com
    User git
    IdentityFile ~/.ssh/id_rsa
    # second                                                                           
    Host second.github.com
    HostName github.com
    User git
    IdentityFile ~/.ssh/id_rsa_x
3. test
ssh -T git@github.com
ssh -T git@second.github.com
4. clone repo to local
git clone git@github.com:xxx/xxx.git
5. unset global name and email, set local name and email
git config --global --unset user.name
git config --global --unset user.email
git consig user.name "xxx"
git consig user.name "xxx@xxx.com"
6. rebuild origin
git remote rm origin
git remote add origin git@second.github.com:xxx/xxxx.git
7. test
git push origin master
